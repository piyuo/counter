import 'dart:async';
import 'package:libcli/module.dart';
import 'package:counter/src/signin/signin_reducer.dart';
import 'package:flutter/material.dart';

class SigninProvider extends PageProvider {
  SettingProvider() {
    redux = Redux(reducer, {
      'name': '',
      'password': '',
    });
  }

  @override
  Future<void> load(BuildContext context) async {}

  Future<bool> submit(BuildContext ctx, String userName) async {
    await dispatch(ctx, SigninAction(name: userName, password: ''));
    return state['name'].isNotEmpty;
  }

  bool get busy {
    return false;
  }
}
